## 文件读取 FileReader

```javascript
let fileInput = document.querySelector('input')

fileInput.onchange = function () {
    // files  文件列表
    let file = fileInput.files[0]

    // FileReader  读取文件
    let reader = new FileReader()

    /**
     * readAsText(file)         读取文本文件 返回字符串形式的内容
     * readAsDataURL(file)      读取图片文件 返回Base64格式
     *
     * readAsArrayBuffer()      返回ArrayBuffer数据对象
     * readAsBinaryString()     返回二进制数据
     */

    reader.readAsDataURL(file)

    /**
     * onload         读取文件成功
     * onabort        读取操作被终止
     * onerror        读取操作发生错误
     * onloadstart    读取文件之前
     * onprogress     读取的过程
     */

    // 文件读取成功
    reader.onload = function () {
        console.log(reader.result)
    }
}
```

## 多媒体播放器

```javascript
let video = document.querySelector('video')

// 浏览器可以播放视频
video.oncanplay = function () {
    /**
     * video.duration           返回视频的时间长度
     * video.playbackRate = 2   控制播放速度
     *
     * video.play()        播放
     * video.pause()       暂停播放
     *
     * video.requestFullscreen()    全屏
     * document.exitFullscreen()       取消全屏
     */

    onBtn.onclick = function () {
        video.play()
    }

    offBtn.onclick = function () {
        video.pause()
    }
}

// 进度 播放
video.ontimeupdate = function () {
    // 已播放的时间
    console.log(video.currentTime)
}

video.onended = function () {
    video.currentTime = 0

    console.log('视频播放完了')
}
```
