# Symbol

#### 1. for of 遍历对象 (Symbol.iterator)

```javascript
let obj = {
    a: 10,
    objArr: ['a', 'b', 'c'],

    [Symbol.iterator]() {
        // 申明变量
        let index = 0

        return {
            next: () => {
                let result = null

                if (index < Object.keys(this).length) {
                    Object.keys(this).forEach((item, i) => {
                        if (index === i) {
                            result = { value: this[item], done: false }
                        }
                    })

                    index++

                    return result
                } else {
                    return { value: undefined, done: true }
                }
            }
        }
    }
}

for (let item of obj) {
    console.log(item)
}
```

# 递归

```javascript
// 深拷贝(递归应用)
function DeepCopy(newObj, obj) {
    for (let k in obj) {
        let item = obj[k]

        if (item instanceof Array) {
            newObj[k] = []

            DeepCopy(newObj[k], item)
        } else if (item instanceof Object) {
            newObj[k] = {}

            DeepCopy(newObj[k], item)
        } else {
            newObj[k] = item
        }
    }

    return newObj
}
```

    console.log('123')

# 正则表达式

#### 1. 字符类

```txt
[a-z]                   匹配方括号之间的任何字符

^[a-z]$                 只能有其中一个字符

^[^0-3]$                ^ 取反, 匹配括号之外的字符
```

#### 2. 量词类

```txt
a*                      匹配任何包含 零个或多个 a 的字符串

a?                      匹配任何包含零个或一个 a 的字符串

a+                      匹配任何包含至少一个 a 的字符串

{n}                     匹配 n次

{n,}                    匹配 n次 或 更多次

{n, m}                  匹配 n ~ m次
```

#### 3. 预定义类

```txt
.*                      匹配任意字符

[\u4e00-\u9fa5]         匹配中文

\d                      查找数字 匹配 0-9 之间的任一字符 (相当于 [0-9])

\D                      匹配 0-9 以外字符 (相当于 [^0-9])

\w                      匹配任意 字母,数字,下划线 (相当于 [a-zA-Z0-9_])

\W                      匹配所有 字母,数字,下划线以外的字符 (相当于 [^a-zA-Z0-9_])

\s                      查找空白字符 匹配空格 (相当于 [\t\r\n\v\f])

\S                      匹配非空格字符 (相当于 [^\t\r\n\v\f])

\b	                    匹配单词边界

\uxxxx	                查找以十六进制数 xxxx 规定的 Unicode 字符
```

#### 4. 全局匹配

```txt
g	                    全局匹配

i	                    忽略大小写

gi                      全局匹配 且 忽略大小写

m	                    执行多行匹配
```

    console.log(e)

 * application/json
 * application/x-www-form-urlencoded:  发送之前编码所有字符
 * multipart/form-data:  文件上传必须使用, 不对字符编码, 数据放在 请求体 中
 * ...
 */

// 发起请求
xhr.send(data)

xhr.onreadystatechange = () => {
    /**
     * readyState
     * 0: XMLHttpRequest对象已创建, 未调用open方法
     * 1: open方法被调用了
     * 2: send方法被调用了, 响应头已接收
     * 3: 数据接受中, response 已经包含部分数据
     * 4: 请求完毕, 数据传输完成或失败

    if (xhr.readyState === 4 && xhr.status === 200) {
        console.log(xhr.responseText)
    }
}

# 手写 Promise

```javascript
function MyPromise(drive) {
    let self = this

    // pending 等待     fulfilled 完成     rejected 失败
    this.status = 'pending' // 状态

    this.value = null // 成功的值
    this.reason = null // 失败的原因

    // 存储  成功,失败的回调
    this.onFulfilledCallbacks = []
    this.onRejectedCallbacks = []

    // 成功的 回调
    function resolve(value) {
        if (self.status === 'pending') {
            self.value = value
            self.status = 'fulfilled'

            self.onFulfilledCallbacks.forEach(item => item(value))
        }
    }

    // 失败的回调
    function reject(reason) {
        if (self.status === 'pending') {
            self.reason = reason
            self.status = 'fulfilled'

            self.onRejectedCallbacks.forEach(item => item(reason))
        }
    }

    try {
        // promise 立即执行
        drive(resolve, reject)
    } catch (err) {
        reject(err)
    }
}

MyPromise.prototype.then = function (onFulfilled, onRejected) {
    onFulfilled =
        typeof onFulfilled === 'function'
            ? onFulfilled
            : function (data) {
                  resolve(data)
              }

    onRejected =
        typeof onRejected === 'function'
            ? onRejected
            : function (err) {
                  throw err
              }

    // 处于等待状态   存储异步任务
    if (this.status === 'pending') {
        this.onFulfilledCallbacks.push(onFulfilled)
        this.onRejectedCallbacks.push(onRejected)
    }
}

let demo = new MyPromise((resolve, reject) => {
    console.log('wzh')

    setTimeout(() => {
        resolve(123)
    }, 0)
})

demo.then(data => {
    console.log(data)
})
```
