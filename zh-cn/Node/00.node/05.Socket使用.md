## Socket 使用

#### 1. WebSocket (ws)

```text
npm install ws
```

```javascript
const WebSocket = require('ws')

// port: 9000   监听的端口号
const wss = new WebSocket.Server({ port: 9000 })

wss.on('connection', ws => {
    // 上线
    ws.on('open', () => {
        console.log('connected')

        ws.send('hello client')
    })

    ws.on('message', data => {
        // ws.send(data)   广播消息给所有客户端
        wss.clients.forEach(client => {
            if (client.readyState === WebSocket.OPEN) {
                client.send(data)
            }
        })
    })

    wss.on('close', () => {
        // 下线
        console.log('disconnected')
    })
})
```

#### 2. socket.io

```
npm install socket.io                服务端
```

```javascript
// 服务端
const express = require('express')
const http = require('http')
const socketIO = require('socket.io')
const app = express()
const server = http.createServer(app)

const io = socketIO(server, {
    // 解决跨域问题
    cors: {
        origin: '*'
    },
    methods: ['GET', 'POST']
})

//连接建立中
io.on('connection', socket => {
    console.log('一个用户与服务器建立连接', socket.handshake.query.id)

    socket.on('sendMsg', data => {
        console.log(`收到客户端的消息：${data}`)

        // 广播消息给所有客户端
        io.emit('pushMsg', '服务端向客户端推送消息...')
    })

    // 监听客户端断开连接事件
    socket.on('disconnect', () => {
        console.log('user disconnected')
    })

    // 断开连接
    // socket.disconnect()
})

server.listen(24000, () => {
    console.log('启动端口24000')
})
```

#### 3. SockJs

```
npm install socket.io                服务端
```

```javascript
// 服务端

// 引入 SockJS 和 http 模块
var sockjs = require('sockjs') // 需要下载 sockjs
var http = require('http')

// 创建 http 服务器
var server = http.createServer()

// 创建 SockJS 服务器
var sockServer = sockjs.createServer()

// 客户端连接时的回调函数
sockServer.on('connection', function (conn) {
    console.log('Client connected')

    // 客户端关闭连接时的回调函数
    conn.on('close', function () {
        console.log('Client disconnected')
    })

    // 接收客户端消息时的回调函数
    conn.on('data', function (message) {
        console.log('Received message: ' + message)

        // 向客户端发送消息
        conn.write('You said: ' + message)
    })
})

// 将 SockJS 服务器附加到 http 服务器
sockServer.installHandlers(server, { prefix: '/chat' })

// 启动 http 服务器
server.listen(8080, function () {
    console.log('Server started')
})
```
