#### 1. path 对路径的处理

```javascript
const path = require('path')

// path.join()     path.resolve()    拼接路径

const pathStr1 = path.join(__dirname, '/01.fs的使用.js')
const pathStr2 = path.resolve(__dirname, '01.fs的使用.js') // pathStr1 === pathStr2

// path.basename()   得到路径的最后一段
path.basename(pathStr1) // 01.fs的使用.js

// path.extname()    得到文件的扩展名
path.extname(pathStr1) // .js
```
