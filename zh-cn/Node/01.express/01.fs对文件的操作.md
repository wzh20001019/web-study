#### 1. fs 的基本使用

```javascript
const fs = require('fs') // 导入fs系统模块

// 读取文件      文件路径   默认编码
fs.readFile('./01.txt', 'utf-8', (err, result) => {
    // err == null, 读取成功
    if (!err) {
        return console.log(result)
    }

    return console.log(err.message)
})

// 写入内容
fs.writeFile('./01.txt', '写入的内容', 'utf-8', err => {
    if (!err) {
        return console.log('写入成功')
    }

    return console.log('写入失败')
})
```
