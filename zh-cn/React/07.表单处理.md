#### 1. 受控组件 表单处理

```javascript
class Test extends React.Component {
    state = {
        text: ''
    }

    handleChange = e => {
        this.setState({
            // [e.target.name]: e.target.value
            text: e.target.value
        })
    }

    render() {
        // onChange方法配合使用
        return (
            <input type="text" name="text" value={this.state.text} onChange={this.handleChange} />
        )
    }
}

ReactDom.render(<Test />, document.querySelector('#root'))
```

#### 2. 非受控组件 表单处理

```javascript
class Test extends React.Component {
    constructor() {
        super()

        // 创建 ref
        this.textRef = React.createRef()
    }

    // 获取文本框的值
    // getValue = () => {
    //     console.log(this.textRef.current.value)
    // }

    render() {
        return (
            <div>
                // 绑定ref, 注意: 不是绑定 value
                <input name="txt" ref={this.textRef} onChange={this.handleChange} />
            </div>
        )
    }
}

ReactDom.render(<Test />, document.querySelector('#root'))
```
