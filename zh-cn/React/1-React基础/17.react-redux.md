#### 安装 react-redux

```txt
yarn add redux

// 依赖于 redux
yarn add react-redux

// 开发者工具
yarn add redux-devtools-extension

// 支持异步 action
yarn add redux-thunk
```

#### 1. 创建 reducer

```javascript
//   src/store/modules/reducer.js

// 初始化 state
const initState = {
    count: 0
}

let reducer = (state = initState, action: { type: string, value: any }) => {
    switch (action.type) {
        case 'send_action':
            return {
                count: action.value
            }
        default:
            break
    }

    return state
}

export default reducer
```

#### 2. 创建 store

```javascript
//   src/store/index.js

// 创建redux中作为核心的store                 支持异步的中间件    合并reducer
import { legacy_createStore as createStore, applyMiddleware, combineReducers } from 'redux'

//引入支持异步 action
import thunk from 'redux-thunk'

//引入开发者工具
import { composeWithDevTools } from 'redux-devtools-extension'

import reducer_one from './modules/reducer'
// ... 其他 reducer

const reducers = combineReducers({
    reducer_one

    // 其他模块的 reducer
})

const store = createStore(reducers, composeWithDevTools(applyMiddleware(thunk)))

export default store
```

#### 3. main.jsx 中引入

```javascript
// main.jsx

// ... 其他代码

import { Provider } from 'react-redux'
import store from './store/index'

const root = ReactDOM.createRoot(document.getElementById('root'))

root.render(
    <Provider store={store}>
        <Router>
            <App />
        </Router>
    </Provider>
)
```

#### 4. 使用

```typescript
// hook 函数
import { useSelector, useDispatch } from 'react-redux'

// 通过 useSelector 获取状态数据
const { count } = useSelector((state: RootState) => ({
    count: state['reducer_one'].count
}))

// 修改数据  useDispatch
const dispatch = useDispatch()

const handleClick = () => {
    dispatch({ type: 'send_action', value: 2000 })
}
```

```javascript
import React, { Component } from 'react'

// 1. connect
import { connect } from 'react-redux'

class Home extends Component {
    handleClick = () => {
        // 4. 发送 dispatch
        this.props.sendAction()
    }

    render() {
        return (
            <>
                <span>{this.props.count}</span>
                <button onClick={this.handleClick}>click</button>
            </>
        )
    }
}

// state映射
const mapStateToProps = state => {
    return {
        count: state.count
    }
}

// dispatch映射
const mapDispatchToProps = dispatch => {
    return {
        sendAction: () => {
            dispatch({
                type: 'send_action'
            })
        }
    }
}

// connect(state映射, dispatch映射)(当前组件类)
export default connect(mapStateToProps, mapDispatchToProps)(Home)
```
