#### 1. 路由的基本使用

```txt
// 1. 下载路由包
npm install react-router-dom
```

```javascript
// 2. 引入Router, Route, Link
import { BrowserRouter as Router, Route, Link } from 'react-router-dom'

class Com extends React.PureComponent {
    render() {
        return (
            // 3. 使用 Router 组件包裹整个应用
            <Router>
                <div>
                    // 4. Link 组件(路由入口) 去往的地址 to="/" 默认路由
                    <Link to={'/'}>路由1</Link>
                    <Link to={`/two${params}`}>传递 params</Link>
                    <Link to={`/three/?id=${id}&title=${title}`}>传递 query</Link>
                    <div>
                        // 5. Route 组件(路由出口) 展示对应的组件 路由规则(path)
                        <Route exact path="/" component={First}></Route>
                        // 6. exact(路由精准匹配)
                        <Route exact path="/two" component={Two}></Route>
                    </div>
                </div>
            </Router>
        )
    }
}

const First = () => {
    return <div>路由1对应内容</div>
}

const Two = () => {
    return <div>路由2对应内容</div>
}

ReactDom.render(<Com a={10} />, document.querySelector('#root'))
```
