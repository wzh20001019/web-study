#### 1. JSX 注意点

```txt
// 1. 特殊属性名
class -> className    for -> htmlFor

// 2. JSX可以用 () 包裹
```

#### 2. JSX 中使用 JavaScript 表达式

```javascript
import React from 'react'
import ReactDom from 'react-dom'

const content = '123'
const jsx1 = <div>jsx</div>

// {} 中不能使用对象, 不能使用if/for
const title = (
    <div className="qw">
        // 使用其他的JSX ...
        {content}
        {'a'}
        {1 + 7}
        {3 > 5 ? 'true' : 'false'}
        {jsx1}
    </div>
)

ReactDom.render(title, document.querySelector('#root'))
```

#### 3. JSX 条件渲染

```javascript
const isLoading = true

const loading = () => {
    if (isLoading) {
        return <div>loading...</div>
    }
    return <div>加载完成</div>
}

const title = <div className="qw">{loading()}</div>

ReactDom.render(title, document.querySelector('#root'))
```

#### 4. JSX 列表渲染

```javascript
const list = [
    {
        id: 1,
        name: 'qwe'
    },
    {
        id: 2,
        name: 'rty'
    },
    {
        id: 3,
        name: 'uio'
    }
]

const title = (
    <div className="qw">
        // 使用map遍历数据, 绑定key -> key={item.id}
        {list.map(item => (
            <div key={item.id}>{item.name}</div>
        ))}
    </div>
)

ReactDom.render(title, document.querySelector('#root'))
```
