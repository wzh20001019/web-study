#### 1. state 的基本使用

```javascript
class Test extends React.Component {
    /* constructor() {
		super()

		this.state = {
			count: 0
		}
	} */

    // 简化写法
    state = {
        count: 0
    }

    render() {
        return <div>{this.state.count}</div>
    }
}

ReactDom.render(<Test />, document.querySelector('#root'))
```

#### 2. setState()修改状态

```javascript
class Test extends React.Component {
    state = {
        count: 0
    }

    customFn() {
        // 修改state中的值, 更新ui
        /* this.setState({
			count: this.state.count + 1
		}) */

        this.setState((state, props) => {
            console.log(state)
            console.log(props)
            return {
                count: state.count + 1
            }
        })
    }

    render() {
        return <div onClick={() => this.customFn()}>{this.state.count}</div>
    }
}

ReactDom.render(<Test />, document.querySelector('#root'))
```

#### 3. setState()进阶

```javascript
// 1. state, props 接收最新值
this.setState((state, props) => {
    console.log(state)
    console.log(props)
    return {
        count: state.count + 1
    }
})

// 2. 状态更新之后的回调函数  setState(updater, callback)
this.setState(
    (state, props) => {
        return {
            a: state.a + 1
        }
    },
    () => {
        console.log('状态更新了')
    }
)
```
