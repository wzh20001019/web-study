#### Vue3 体验

```html
<div class="app">
    <span :class="{cls}">{{message}}</span>
    <!-- cls是一个类名 -->
    <ul>
        <li v-for="item in list" :key="item.id" :title="item.name" @click="fn(item.id)">
            {{item.id}}---{{itemname}}
        </li>
    </ul>
</div>
```

```javascript
<script>
const app = Vue.createApp({
    data() {
        return {
            message: 'Vue初体验',
            list: [
                { id: 1, name: '张三' },
                { id: 2, name: '李四' },
                { id: 3, name: '王五' }
            ],
            cls: true
        }
    },
    methods: {
        fn(id) {
            console.log(id)
        }
    }
})

const vm = app.mount('.app')
</script>
```
