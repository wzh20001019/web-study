#### 父组件 绑定自定义属性传值

```vue
<template>
    <son :msg="msg"></son>
    <!-- 在子组件上自定义属性 绑定父组件的值 -->
    <son :arr="arr"></son>
    <!-- 可以拿到数组 -->
</template>

<script>
import son from './son.vue' //导入子组件模块

export default {
    name: 'father',
    data() {
        return {
            msg: '父组件的值'
        }
    },
    components: {
        son //挂载子组件
    }
}
</script>

<style scoped></style>
```

#### 子组件 props 获取自定义的属性从而得到值

```vue
<template>
    <div>{{ msg }}</div>
    <!-- 通过props把父组件的值插入元素中 -->
    <div>
        <div v-for="item in arr" :key="item.id">{{ item }}</div>
    </div>
</template>

<script>
export default {
    name: 'son',
    data() {
        return {}
    },

    //获取父组件上绑定的值
    props: {
        msg: {
            type: String, // 定义获取的数据类型
            default: '填充值', //获取msg值失败的话显示 default里的值
            required: true //msg 这个值是父组件必传的值, 不传会报错
        },
        arr: {
            type: Array,
            default: ['a', 'b', 'c']
        }
    }
}
</script>

<style scoped></style>
```
