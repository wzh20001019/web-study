## 封装网络请求

```javascript
import axios from 'axios' //引入 axios

import store from '@/store' //引入 store 文件
```

## 全局设置

```javascript
//全局设置请求时间(超时)
axios.defaults.timeout = 3000

//全局设置根目录
axios.defaults.baseURL = 'http://127.0.0.1'

//全局设置请求头
axios.defaults.header = {
    'Content-Type': 'application/x-www-form-urlencoded'
}
```

## create 一个实例

```javascript
const request = axios.create({
    //设置根目录 (基准路径)
    baseURL: 'http://127.0.0.1',

    //设置请求时间(超时)
    timeout: 5000
})
```

## request (请求拦截)

```javascript
//请求拦截
request.interceptors.request.use(
    config => {
        const { user } = store.state //解构保存在 vuex 中的user对象

        //判断是否登录,同意配置请求头
        if (user && user.token) {
            config.headers.Authorization = `Bearer ${user.token}`
        }

        return config //放行配置信息
    },
    error => {
        return Promise.reject(error) //捕获请求错误信息
    }
)
```

## response (响应拦截)

```javascript
//响应拦截
request.interceptors.response.use(
    results => {
        //results = results.data
        return results
    },
    error => {
        //响应对应的错误
        Promise.reject(error)

        // err.response.data.errors[Object.keys(err.response.data.errors)[0]][0]
    }
```

## 共享实例

```javascript
export default request
```

## 使用封装的实例 (封装接口)

```javascript
//引入封装的axios文件
import request from '@/api/...'

export const login = data => {
    //导出封装的请求函数
    return request({
        method: 'POST',
        url: '/api/...',
        data
    })
}

export const get = query => {
    //导出封装的请求函数
    return request({
        method: 'GET',
        url: '/api/...',
        params: query
    })
}
```

## 使用封装的接口

```javascript
import { login } from './request.js'


methods: {
	async fn() {
		try {
			const res = await login(data)
			console.log(res)
		} catch (err) {
			console.log(err.message)
		}
	}
}
```

## axios 封装

```javascript
import axios from 'axios'
import store from '@/plugins/store/index'

import { Loading, MessageBox } from 'element-ui'

const request = axios.create({
    // 设置请求根目录
    baseURL: 'http://127.0.0.1:8080',

    // 设置请求超时
    timeout: 10000
})

// 请求数据时的加载动画
let loadingInstance = null

// 设置请求拦截
request.interceptors.request.use(
    // 请求前的配置操作
    config => {
        // 获取本地储存的 token
        const token = localStorage.getItem('token')

        // 配置token
        if (token) {
            config.headers.Authorization = 'Bearer ' + token
        }

        // 请求时加载动画
        loadingInstance = Loading.service({
            fullscreen: true,
            // 动画中的文字
            text: '加载中',
            spinner: 'el-icon-loading',
            background: 'rgba(0, 0, 0, 0.7)'
        })

        return config
    },

    // 请求失败
    error => {
        // 捕获错误信息
        return Promise.reject(error)
    }
)

// 设置响应拦截
request.interceptors.response.use(
    // 对获取的数据进行处理
    results => {
        // if (results.code === 1) {
        //   // 关闭加载动画
        //   //

        //   // 返回数据
        //   //   return results.data

        //   return results
        // }

        if (results) {
            loadingInstance.close()
        }

        if (results.data.code !== 200) {
            return MessageBox.alert(results.data.msg, '提示', {
                confirmButtonText: '确定',
                callback: () => {
                    console.log('response: ' + '错误')
                }
            })
        }

        return results.data
    },

    // 获取数据失败
    error => {
        console.log(error)

        // MessageBox.alert(error.message, '提示', {
        //   confirmButtonText: '确定',
        //   callback: () => {
        //     // console.log('无权限')
        //   }
        // })

        if (error) {
            loadingInstance.close()
        }

        if (error.response) {
            switch (error.response.status) {
                // 需要用户验证
                case 401:
                    MessageBox.alert('无权限', '提示', {
                        confirmButtonText: '确定',
                        callback: () => {
                            // console.log('无权限')
                        }
                    })

                    // 这里先做清空 token操作
                    // store.state.token = ''
                    // localStorage.removeItem('token')

                    break

                // 链接问题
                case 404:
                    MessageBox.alert('请求地址错误', '提示', {
                        confirmButtonText: '确定',
                        callback: () => {
                            // console.log('请求地址错误')
                        }
                    })

                    break

                // 请求超时处理
                case 408:
                    MessageBox.alert('请求超时', '提示', {
                        confirmButtonText: '确定',
                        callback: () => {
                            // console.log('请求超时')
                        }
                    })

                    break

                // 内部服务器错误
                case 500:
                    MessageBox.alert('内部服务器错误', '提示', {
                        confirmButtonText: '确定',
                        callback: () => {
                            // console.log('内部服务器错误')
                        }
                    })

                    break

                default:
                    MessageBox.alert('其他错误', '提示', {
                        confirmButtonText: '确定',
                        callback: () => {
                            // console.log('请求参数错误')
                        }
                    })
            }
        }
        return Promise.reject(error)
    }
)

export default request
```
