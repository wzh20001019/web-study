#### Vue3 的生命周期

```txt
beforeCreate() {
	console.log('实例刚刚被创建')
},

created() {
	console.log('实例已经创建完成')
},


beforeMount() {
	console.log('模板编译之前')
},

mounted() {
	console.log('模板编译完成')
},


beforeUpdate() {
	console.log('数据更新之前')
},

updated() {
	console.log('数据更新完成')
},


activated() {
	console.log('keep-alive 缓存的组件激活时使用')    //<keep-alive></keep-alive>
},

deactivated() {
	console.log('keep-alive 缓存的组件停用时使用')
},


beforeUnmount() {
	console.log('实例销毁之间')
},

unmounted() {
	console.log('实例销毁完成')
}
```

#### nextTick (延迟处理)

```javascript
//延迟处理(将回调延迟到下次DOM循环更新之后执行, 在修改数据之后立即使用它,然后等待DOM更新)
this.$nextTick(() => {
    //执行代码
})
```
