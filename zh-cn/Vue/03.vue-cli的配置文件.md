#### 根目录创建 vue.config.js 可以自己修改配置

```javascript
import { resolve } from 'path'

module.exports = {
    outputDir: 'build',   //自定义打包出口文件夹名称  (默认dist)

    //可使用webpack的配置 (内嵌到vue配置中)
    configureWebpack: {
    	plugins: [],

        resolve: {
            alias: {
                'assets': 'resolve(__dirname, 'assets')'    //自定义文件别名
            }
        }
    },
    publicPath: './'
}
```
