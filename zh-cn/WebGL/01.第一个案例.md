```javascript
import './lib/OrbitControls.js'

window.onload = init

function init() {
  //* 创建场景
  let scene = new THREE.Scene()

  //* 设置摄像机  (角度, 长宽比, 最近可以看到, 最远可以看到)
  let camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 2000)

  //* 创建渲染器
  let renderer = new THREE.WebGLRenderer()

  // 清除颜色
  renderer.setClearColor(new THREE.Color('#111'))

  //* 输出 canvas 画面的大小
  renderer.setSize(window.innerWidth, window.innerHeight)

  // 设置渲染物体阴影
  renderer.shadowMap.enabled = true

  // 显示三维坐标系     红: x    绿: y    蓝: z
  let axes = new THREE.AxesHelper(30)

  // 添加坐标系到场景中
  //   scene.add(axes)

  // 创建地面的几何体 (宽, 高)
  let planeGeometry = new THREE.PlaneGeometry(50, 20)

  // 给地面物体上色
  let planeMaterial = new THREE.MeshStandardMaterial({ color: '#ccc' })

  // 创建地面 (地面的几何体, 物体上色)
  let plane = new THREE.Mesh(planeGeometry, planeMaterial)

  // 物体移动位置
  plane.rotation.x = -0.5 * Math.PI
  plane.position.x = 0
  plane.position.y = 0
  plane.position.z = 0

  plane.castShadow = true
  plane.receiveShadow = true

  // 将地面添加到场景中
  scene.add(plane)

  // 添加立方体  (长, 宽, 高)
  let cubeGeometry = new THREE.BoxGeometry(4, 4, 4)
  let cubeMaterial = new THREE.MeshLambertMaterial({
    color: '#74F9FF'
  })
  let cube = new THREE.Mesh(cubeGeometry, cubeMaterial)

  //   cube.rotation.x = -0.5 * Math.PI
  cube.position.x = 0
  cube.position.y = 4
  cube.position.z = 0

  // 对象是否渲染到 阴影贴图当中
  cube.castShadow = true

  scene.add(cube)

  // 创建球体
  let sphereGeometry = new THREE.SphereGeometry(2, 20)
  let sphereMaterial = new THREE.MeshLambertMaterial({
    color: '#9896F1'
  })
  let sphere = new THREE.Mesh(sphereGeometry, sphereMaterial)

  sphere.position.x = 8
  sphere.position.y = 4
  sphere.position.z = 0

  sphere.castShadow = true

  scene.add(sphere)

  // 创建聚光灯
  let spotLight = new THREE.SpotLight('#fff')
  spotLight.position.set(120, 120, -20)
  spotLight.castShadow = true

  spotLight.angle = Math.PI / 10
  spotLight.shadow.penumbra = 0.05
  spotLight.shadow.mapSize.width = 1024
  spotLight.shadow.mapSize.height = 1024

  scene.add(spotLight)

  //* 摄像机定位, 指向场景中心
  camera.position.x = 40
  camera.position.y = 40
  camera.position.z = 40

  // 摄像机朝向指向 场景中心
  camera.lookAt(scene.position)

  //* 将渲染器输出添加到 html 元素中
  document.querySelector('#webgl-output').appendChild(renderer.domElement)

  //* 渲染
  renderer.render(scene, camera)

  let controls = new THREE.OrbitControls(camera, renderer.domElement)

  controls.addEventListener('change', () => {
    renderer.render(scene, camera)
  })

  function render() {
    renderer.render(scene, camera)

    cube.rotateY(0.01)
    cube.rotateX(0.01)
    cube.rotateZ(0.01)
    window.requestAnimationFrame(render)
  }

  window.requestAnimationFrame(render)
}
```

```javascript
import * as THREE from 'three'
import { RGBELoader } from 'three/examples/jsm/loaders/RGBELoader'
import { OrbitControls } from 'three/examples/jsm/controls/OrbitControls' // 控制器
import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader' // 模型解析器

class Base3d {
  constructor(selector) {
    this.container = document.querySelector(selector)

    this.scene // 场景
    this.camera // 摄像机
    this.renderer // 渲染器
    this.model // 模型

    this.init()

    this.animation()
  }

  init() {
    this.initScene() // 初始化场景

    this.initCamera() // 初始化像机

    this.initRender() // 初始化渲染器

    this.initControls() // 控制器

    this.addMesh() // 添加物体

    window.addEventListener('resize', this.onWindowResize.bind(this))
  }

  // 初始化场景
  initScene() {
    // 创建场景
    this.scene = new THREE.Scene()

    this.setEnvMap('1.hdr')
  }

  // 场景图片
  setEnvMap(hdr) {
    const rgbELoader = new RGBELoader()

    rgbELoader.setPath('files/').load(hdr, texture => {
      // 图片球形展示
      texture.mapping = THREE.EquirectangularReflectionMapping

      // 场景的背景              纹理
      this.scene.background = texture
      // 场景的环境设置          纹理
      this.scene.environment = texture
    })
  }

  // 初始化像机
  initCamera() {
    // 设置摄像机  (角度, 宽高比, 最近可以看到, 最远可以看到的距离)
    this.camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 2000)

    // 摄像机定位, set(x, y, z)    指向场景中心
    this.camera.position.set(-1.8, 0.8, 2.7)

    // this.camera.position.x = -1.8
  }

  // 初始化渲染器
  initRender() {
    // 创建渲染器
    this.renderer = new THREE.WebGLRenderer({
      // 抗锯齿
      antialias: true
    })

    // 设置屏幕像素比
    this.renderer.setPixelRatio(window.devicePixelRatio)

    // 渲染尺寸大小    (画面的大小)
    this.renderer.setSize(window.innerWidth, window.innerHeight)

    // 色调映射
    this.renderer.toneMapping = THREE.ACESFilmicToneMapping
    this.renderer.toneMappingExposure = 2

    // 渲染到html元素上
    this.container.appendChild(this.renderer.domElement)
  }

  // 渲染
  render() {
    // 渲染   render(场景, 摄像机)
    this.renderer.render(this.scene, this.camera)
  }

  animation() {
    this.renderer.setAnimationLoop(this.render.bind(this))
  }

  // 控制器
  initControls() {
    this.controls = new OrbitControls(this.camera, this.renderer.domElement)
  }

  // 模型
  setModel(modelName) {
    return new Promise((resolve, reject) => {
      const loader = new GLTFLoader().setPath('gltf/')
      loader.load(modelName, gltf => {
        console.log(gltf)
        this.model = gltf.scene.children[0]

        // 将模型添加到场景中
        this.scene.add(this.model)

        resolve(modelName + '添加成功')
      })
    })
  }

  // 添加物体
  addMesh() {
    // this.setModel('1.glb')
    // this.setModel('2.glb')
    this.setModel('1.gltf')
  }

  // 监听  动态更新视图
  onWindowResize() {
    this.camera.aspect = window.innerWidth / window.innerHeight
    this.camera.updateProjectionMatrix()
    this.renderer.setSize(window.innerWidth, window.innerHeight)

    // this.render()
  }
}

export default Base3d
```
