#### 1. 储存 arr.length 变量， ++i

```javascript
const list = []
for (let i = 0; i < 10000000; ++i) {
    list.push(i)
}

console.time('方案1')
for (let i = 0; i < list.length; ++i) {
    const value = list[i]
}
console.timeEnd('方案1')

console.time('方案2')
for (let i = 0, length = list.length; i < length; ++i) {
    const value = list[i]
}
console.timeEnd('方案2')

console.time('方案3')
for (let i = list.length - 1; i > -1; --i) {
    const value = list[i]
}
console.timeEnd('方案3')
```
