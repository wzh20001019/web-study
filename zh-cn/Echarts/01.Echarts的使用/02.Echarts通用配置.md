#### 1. title(标题组件)

```json
title: {
    text: '标题',

    // 设置标题样式
    textStyle: {
        color: '#333',
        ...
    },

    borderWidth: number,          // 标题边框
    borderColor: '#ccc',          // 边框颜色
    borderRadius: number,         // 边框圆角
    ...
}
```

#### 2. tooltip(提示框组件) 鼠标 滑过或点击 图表

```json
tooltip: {
    trigger: 'item',              // 触发的类型   item        axis
    triggerOn: 'mousemove',       // 触发的条件   mousemove   click

    // 格式化数据
    // formatter:  '{a} {b}',
    formatter: function(arg) {
        console.log(arg)
    }
}
```

#### 3. toolbox(工具栏组件)

```json
toolbox: {
    feature: {
        saveAsImage: {},             // 导出图片
        dataView: {},                // 数据视图, 可更改数据
        restore: {},                 // 重置, 配置项还原
        dataZoom: {},                // 区域缩放
        magicType: {                 // 动态图表类型切换
            type: ['bar', 'line']
        }
    }
}
```

#### 4. legend(图例组件)

```json
// 和series配合使用
legend: {
    data: ['name1', 'name2']         // series对应的name
}
```
