#### 1. grid(直角坐标系内绘制网格)

```json
grid: {
    show: true,                 // 是否显示网格
    borderWidth: 1,             // 网格的边框线宽
    borderColor: '#ccc',        // 边框线颜色

    top: 10,                    // 组件离容器上侧的距离
    width: 300,                 // 图表的宽度

    ...
}
```

#### 2. axis 配置

```json
xAxis: {
    type: 'category',           // category    value

    // 坐标轴轴线相关设置
    axisLine: {
        // 坐标轴轴线样式配置
        lineStyle:{
            color: '#cc0000',
            ...
        }
    },

    // 坐标轴刻度相关设置
    axisTick: {
        show: true,                     // 是否显示刻度
        alignWithLabel: true,           // 刻度是否在刻度标签中间显示
    },

    // 刻度标签相关配置
    axisLabel: {
        color: '#ccc'
    },

    // 分割线
    splitLine: {
        // 线的样式
        lineStyle:{
            color: '#ccc'
        }
    }
},

yAxis: {
    ...
}
```

#### 3. dataZoom(区域缩放)

```json
dataZoom: [
    {
        type: 'slider',         // slider(滑块)   inside(依靠鼠标滚轮或双指缩放)
        xAxisIndex: 0           // x轴的筛选
    {
        type: 'slider',
        yAxisIndex: 0
    }
]
```
