#### 1. 示例代码

```javascript
let option = {
    color: ['#3AA1FF'],
    grid: {
        show: false, // 是否显示网格

        left: '8%',
        top: '5%',
        right: '5%',
        bottom: '10%'
    },
    tooltip: {
        trigger: 'axis',

        // 触发的条件   mousemove   click
        triggerOn: 'mousemove',

        axisPointer: {
            // shadow(柱形阴影)   line(坐标轴指示)
            type: 'line',

            shadowStyle: {
                color: 'rgba(150,200,200,0.3)'
            }
        },

        textStyle: {
            fontWeight: 100
        }
    },
    xAxis: {
        type: 'category',
        // data: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'],

        // 坐标轴轴线相关设置
        axisLine: {
            // 坐标轴轴线样式配置
            lineStyle: {
                color: '#ccc'
            }
        },
        // 坐标轴刻度相关设置
        axisTick: {
            // 刻度是否在刻度标签中间显示
            alignWithLabel: true,

            length: 4,

            lineStyle: {
                color: '#ccc'
            }
        },

        axisLabel: {
            margin: 10
        }
    },
    yAxis: {
        type: 'value',
        axisLine: {
            show: false
        },

        // 坐标轴刻度相关设置
        axisTick: {
            show: false
        },

        axisLabel: {
            margin: 10
        },
        splitLine: {
            // 隐藏或显示
            show: true,
            // 网格线
            lineStyle: {
                // 设置网格线类型 dotted：虚线   solid:实线
                type: 'dashed'
            }
        }
    },
    series: [
        {
            // data: [120, 200, 150, 80, 70, 110, 130],
            type: 'bar',
            barWidth: 40
        }
    ]
}
```
